snd.dtype

# Plot waveform
timeArray = arange(0, length, 1)#contains sample number (0,1,2,3,...)
timeArray = timeArray / samplerate #contains time label in seconds
timeArray = timeArray * 1000  #scale to milliseconds

plot(timeArray, ch1, color='k')
ylabel('Amplitude')
xlabel('Time (ms)')
